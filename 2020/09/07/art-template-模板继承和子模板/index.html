模板继承

标准语法

{{extend './layout.art'}}
{{block 'head'}} ... {{/block}}

使用方法：

layout.html

<!DOCTYPE html> <html lang="en"> <head>    
    <meta charset="UTF-8">    
    <meta name="viewport" content="width=device-width, initial-scale=1.0">    
    <title>Document</title> 
    </head> 
    <body>    
        {{ include'./header.html' }}

index.html

<!-- 在 index.html 中继承 layout.html -->

<!– 不填坑就显示默认内容 填坑就显示填坑内容 –>


{{ extend ‘./layout.html’ }}


<!– 填坑 –>
{{ block ‘content’ }}
    <div>
        <h1>
            index 页面填坑内容
        </h1>
    </div>
{{ /block }}
